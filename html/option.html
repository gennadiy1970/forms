select

элемент с выбором опций

<select name="select"> 
    <option value="css">CSS</option>
    <option value="html">HTML</option>
    <option value="js">JavaScript</option>
</select>

datalist

элемент содержащий возможные опции для выбора элемента input с атрибутом list

<label for="myBrowser">Выбор браузера</label>
    <input list="browsers" id="myBrowser" name="myBrowser" /> 
    <datalist id="browsers">
        <option>Chrome</option>
        <option>Firefox</option>
        <option>Internet Explorer</option>
        <option>Opera</option>
        <option>Safari</option>
        <option>Microsoft Edge</option>
</datalist>

option

  • option используется для определения пункта списка контейнера select, optgroup и datalist
    • value // не обязательный атрибут. Если отсутствует, то значение берется из текста
    • selected // не обязательный атрибут. Выделяет значение по умолчанию
    • disabled // не обязательный атрибут. Делает опцию недоступной для выделения
    • label // не обязательный атрибут. Если отсутствует, то значение берется из текста

optgroup, select, option

<select>
    <optgroup label="Группа 1">
        <option>Frontend</option>
    </optgroup>
    <optgroup label="Группа 2">
        <option value="CSS">CSS</option>
        <option> selected ;HTML</option>
    </optgroup>
    <optgroup label="Группа 3" disabled>
        <option>JavaScript</option>
        <option>React</option>
        <option>Angular</option>
    </optgroup>
</select>
